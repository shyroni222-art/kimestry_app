Thank you for installing {{ .Chart.Name }}!

Your release "{{ .Release.Name }}" has been deployed.

To access your application:

1. Get the frontend service URL:
{{- if .Values.frontend.service.loadBalancerIP }}
  export FRONTEND_URL={{ .Values.frontend.service.loadBalancerIP }}
{{- else if contains "LoadBalancer" .Values.frontend.service.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
  Watch the status with: kubectl get svc --namespace {{ .Release.Namespace }} -w {{ include "kimestry.frontend.fullname" . }}

  export FRONTEND_URL=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "kimestry.frontend.fullname" . }} -o jsonpath='{.status.loadBalancer.ingress[0].ip}')
{{- else if contains "ClusterIP" .Values.frontend.service.type }}
  export FRONTEND_URL=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  export FRONTEND_PORT=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "kimestry.frontend.fullname" . }} -o jsonpath="{.spec.ports[0].nodePort}")
  echo "Frontend URL: http://$FRONTEND_URL:$FRONTEND_PORT"
{{- end }}

2. Get the backend service URL:
  Backend is available internally at: {{ include "kimestry.backend.fullname" . }}:{{ .Values.backend.service.port }}

3. Access the frontend in your browser at the URL above.

To check the status of your pods:
  kubectl get pods --namespace {{ .Release.Namespace }}

To view logs:
  kubectl logs -l app.kubernetes.io/name={{ include "kimestry.backend.name" . }} --namespace {{ .Release.Namespace }}
  kubectl logs -l app.kubernetes.io/name={{ include "kimestry.frontend.name" . }} --namespace {{ .Release.Namespace }}

{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  - http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}
{{- end }}
{{- end }}
